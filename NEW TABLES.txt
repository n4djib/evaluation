= Create tables classement & classement_year *******************
CREATE TABLE classement (
    id          INTEGER         PRIMARY KEY NOT NULL,
    promo_id    INTEGER,
    student_id  INTEGER,
    avr_licence NUMERIC (10, 2),
    avr_master  NUMERIC (10, 2),
    FOREIGN KEY ( student_id )
    REFERENCES student (id) ON DELETE CASCADE  ON UPDATE CASCADE,
    FOREIGN KEY ( promo_id )
    REFERENCES promo (id) ON DELETE CASCADE  ON UPDATE CASCADE
);

CREATE TABLE classement_year (
    id             INTEGER         PRIMARY KEY
                                   NOT NULL,
    classement_id  INTEGER,
    year           INTEGER,
    average        NUMERIC (10, 2),
    average_app    NUMERIC (10, 2),
    R              NUMERIC (10, 2),
    R_app          NUMERIC (10, 2),
    S              NUMERIC (10, 2),
    S_app          NUMERIC (10, 2),
    avr_classement NUMERIC (10, 2),
    FOREIGN KEY (
        classement_id
    )
    REFERENCES classement (id) 
);
