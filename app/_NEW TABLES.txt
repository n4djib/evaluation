
CREATE TABLE notification (
    id           INTEGER NOT NULL,
    title        VARCHAR (45),
    notification VARCHAR (255),
    complite     BOOLEAN,
    timestamp    DATETIME,
    delete_url   VARCHAR (255),
    user_id      INTEGER,
    PRIMARY KEY (
        id
    )
);
